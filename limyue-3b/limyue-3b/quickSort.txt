template <class T>
void swap (T &a, T &b)
{
	T c = a; a = b; b = c;
}
template <class T>
int partition (T &d, int left, int right)
{
	int i, j, p;
	p = d[left];
	i = left;
	j = right + 1;
	do
	{
		do
		{
			i = i + 1;
		} while (d[i] < p && i < right);
		do
		{
			j = j - 1;
		} while (d[j] > p);
		swap (d[i], d[j]);
	} while (i < j);
	swap (d[i], d[j]);
	swap (d[left], d[j]);
	return j;
}
template <class T>
void quickSort(T &d, int left, int right)
{
	if (left < right)
	{
		int s = partition (d, left, right);
		quickSort(d, left, s - 1);
		quickSort(d, s + 1, right);
	}
}
